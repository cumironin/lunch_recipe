{{< checkbox "1 (11 ounce) can DOLE Mandarin Oranges, drained" >}}
{{< checkbox "1  DOLE Banana, sliced" >}}
{{< checkbox "1 ½ cups seedless grapes" >}}
{{< checkbox "½ cup miniature marshmallows" >}}

{{< checkbox "¼ cup flaked coconut, toasted" >}}


{{< direction >}}
