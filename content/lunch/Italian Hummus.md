{{< boldheading >}}

{{< checkbox "1 bunch fresh basil, chopped" >}}
{{< checkbox "1 pint grape tomatoes, coarsely chopped" >}}
{{< checkbox "1 clove garlic, minced" >}}
{{< checkbox "salt and pepper to taste" >}}


{{< direction >}}
